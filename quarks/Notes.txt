Reference: https://quarkus.io/guides/buildinggu-native-image

Install graalvm
gu native-image

git clone https://github.com/quarkusio/quarkus-quickstarts.git
package
	./mvnw package -Dnative
test
	./mvnw verify -Pnative
build native image
	./mvnw package -Dnative -Dquarkus.native.container-build=true
